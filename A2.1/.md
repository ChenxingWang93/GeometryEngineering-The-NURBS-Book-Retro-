ALGORITHM A2.1
  int FindSpan(n,p,u,U)
    { /*  Determine the knot span index  */
      /*  Input:  n,p,u,U  */
      /*  Return: the knot span index  */
    if (u == U[n+1])  return(n);    /* Special case */
    low = p;  high = n+1;   /* Do binary search */
    mid = (low+high)/2;
    while (u < U[mid] || u >= U[mid+1])
  {
  if (u < U[mid])   high = mid;
    else            low = mid;
  mid = (low+high)/2
  }
    return(mid);
    }
