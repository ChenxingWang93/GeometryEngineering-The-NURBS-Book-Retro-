  if (u >= U[i+j] && u < U[i+j+1])   N[j] = 1.0;

  {
  if (N[0] == 0.0)  saved = 0.0;

  for (j=0; j<p-k+1; j++)
  {
  Uleft = U[i+j+1];
  Uright = U[i+j+k+1];
  if (N[j+1] == 0.0)

    N[j] = saved;   saved = 0.0;
    }
    else

    }
  }
}
